h1. Other Selectors  


{% highlight javascript %}

x$(window);                         // The Window
x$(element);                        // An Existing Element  
x$('ul#globalnav li a.selected');   // A CSS3 Selector
x$(['li','div#foo']);               // An Array of Selectors
x$('li','.selected','#some_id');    // A comma list of CSS3 Selectors

{% endhighlight %}
